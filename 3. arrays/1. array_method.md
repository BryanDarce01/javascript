# SLICE

``` javascript

const arreglo = ['a', 'b', 'c', 'd', 'e'];
console.log(arreglo.slice(2)); 

// ['c', 'd', 'e']

console.log(arreglo.slice(2, 4)); 

// ['c', 'd']

console.log(arreglo.slice(-1)); 

// ['e']


// ESTOS ÚLTIMOS NOS DEVUELVE UNA COPIA DEL ARREGLO ORIGINAL
console.log(arreglo.slice()); // ['a', 'b', 'c', 'd', 'e']
console.log(...arreglo()); // a b c d e
console.log([...arreglo]); // ['a', 'b', 'c', 'd', 'e']

```

# SPLICE

Este método si modifica el arreglo original

``` javascript

const arreglo = ['a', 'b', 'c', 'd', 'e'];

console.log(arreglo.splice(2)); // ['c', 'd', 'e']

console.log(arreglo); // ['a', 'b']

arreglo.splice(1,2); // Elimina 2 elementos a partir del índice 1
console.log(arreglo); // Resultado: ['a', 'd', 'e']

```

# REVERSE

Este método también modifica el arreglo original

``` javascript

const arreglo = ['a', 'b', 'c', 'd', 'e'];

console.log(arreglo.reverse()); // ['e', 'd', 'c', 'b', 'a']

```

# CONCAT

Este método no modifica al arreglo original

``` javascript

const arreglo = ['a', 'b', 'c', 'd', 'e'];
const arreglo2 = ['f', 'g', 'h', 'i', 'j'];

console.log(arreglo.concat(arreglo2)); 
// ['a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j']


// NOS DA EL MISMO RESULTADO CON EL OPERADOR SPREAD

console.log([...arreglo, ...arreglo2]); 
// ['a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j']
```

# JOIN

``` javascript

const arreglo = ['a', 'b', 'c', 'd', 'e'];

console.log(arreglo.join(' - ')); // 'a - b - c - d - e'

```
